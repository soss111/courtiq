Google Sheets sünkroonib kolme eraldi lehte: Kanded, Mängijad, Harjutused.
Kuidas seadistada Apps Script? ▾
1. Ava sheets.google.com → loo uus tabel (lehtede nimed pole olulised — skript loob need ise ära)
2. Ava Laiendused → Apps Script
3. Kustuta vana kood, kleebi see kood:
4. Vali Juuruta → Uus juurutus → Veebirakendus, sea juurdepääs "Kõigile"
5. Kopeeri URL siia ja vajuta "Testi ühendust"
Oluline: kui muudad koodi hiljem, vali alati Uus juurutus (mitte "Manage deployments"), muidu vana URL ei kasuta uut koodi.

function getSheet(name, headers) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.appendRow(headers);
  }
  return sh;
}

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    const type = d.type || 'session';

    if (type === 'session') {
      const sh = getSheet('Kanded', ['Kuupäev','Mängija','Harjutus','Kestus','Intensiivsus','Märkmed','Salvestatud']);
      sh.appendRow([d.kuupäev, d.mängija, d.harjutus, d.kestus, d.intensiivsus, d.märkmed||'', new Date().toISOString()]);
    } else if (type === 'player') {
      const sh = getSheet('Mängijad', ['ID','Nimi','Positsioon','Vanus','Uuendatud']);
      const rows = sh.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === d.id) {
          sh.getRange(i+1,1,1,5).setValues([[d.id,d.nimi,d.positsioon,d.vanus,new Date().toISOString()]]);
          found = true; break;
        }
      }
      if (!found) sh.appendRow([d.id, d.nimi, d.positsioon, d.vanus, new Date().toISOString()]);
    } else if (type === 'drill') {
      const sh = getSheet('Harjutused', ['ID','Nimi','Ikoon','Kategooria','Tase','Kestus','Seeriad','SeeriaSek','Puhkus','Kirjeldus','Sammud','Vead','Aktiivne','Uuendatud']);
      const rows = sh.getDataRange().getValues();
      const rowData = [d.id, d.nimi, d.ikoon||'🏀', d.kategooria, d.tase, d.kestus, d.seeriad||3, d.seeriaSek||30, d.puhkus||15, d.kirjeldus||'', JSON.stringify(d.sammud||[]), JSON.stringify(d.vead||[]), d.aktiivne, new Date().toISOString()];
      let found = false;
      for (let i = 1; i < rows.length; i++) {
        if (rows[i][0] === d.id) {
          sh.getRange(i+1,1,1,rowData.length).setValues([rowData]);
          found = true; break;
        }
      }
      if (!found) sh.appendRow(rowData);
    } else if (type === 'batch') {
      const items = d.items || [];
      for (const item of items) {
        const t = item.type;
        if (t === 'session') {
          const sh = getSheet('Kanded', ['Kuupäev','Mängija','Harjutus','Kestus','Intensiivsus','Märkmed','Salvestatud']);
          sh.appendRow([item['kuupäev'], item['mängija'], item.harjutus, item.kestus, item.intensiivsus, item['märkmed']||'', new Date().toISOString()]);
        } else if (t === 'player') {
          const sh = getSheet('Mängijad', ['ID','Nimi','Positsioon','Vanus','Uuendatud']);
          const rows = sh.getDataRange().getValues();
          let found = false;
          for (let i = 1; i < rows.length; i++) {
            if (rows[i][0] === item.id) {
              sh.getRange(i+1,1,1,5).setValues([[item.id,item.nimi,item.positsioon,item.vanus,new Date().toISOString()]]);
              found = true; break;
            }
          }
          if (!found) sh.appendRow([item.id, item.nimi, item.positsioon, item.vanus, new Date().toISOString()]);
        } else if (t === 'drill') {
          const sh = getSheet('Harjutused', ['ID','Nimi','Ikoon','Kategooria','Tase','Kestus','Seeriad','SeeriaSek','Puhkus','Kirjeldus','Sammud','Vead','Aktiivne','Uuendatud']);
          const rows = sh.getDataRange().getValues();
          const rowData = [item.id, item.nimi, item.ikoon||'🏀', item.kategooria, item.tase, item.kestus, item.seeriad||3, item.seeriaSek||30, item.puhkus||15, item.kirjeldus||'', JSON.stringify(item.sammud||[]), JSON.stringify(item.vead||[]), item.aktiivne, new Date().toISOString()];
          let found = false;
          for (let i = 1; i < rows.length; i++) {
            if (rows[i][0] === item.id) {
              sh.getRange(i+1,1,1,rowData.length).setValues([rowData]);
              found = true; break;
            }
          }
          if (!found) sh.appendRow(rowData);
        }
      }
    } else if (type === 'delete') {
      const sheetName = d.kind === 'player' ? 'Mängijad' : d.kind === 'drill' ? 'Harjutused' : null;
      if (sheetName) {
        const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
        if (sh) {
          const rows = sh.getDataRange().getValues();
          for (let i = rows.length - 1; i >= 1; i--) {
            if (rows[i][0] === d.id) {
              sh.deleteRow(i + 1);
              break;
            }
          }
        }
      }
    }

    return ContentService.createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false, error:String(err)}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function sheetToObjects(sh) {
  const rows = sh.getDataRange().getValues();
  if (rows.length < 2) return [];
  const headers = rows[0];
  return rows.slice(1).map(r => {
    const o = {};
    headers.forEach((h, i) => o[h] = r[i]);
    return o;
  });
}

function doGet(e) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const out = {ok: true};
    const kanded = ss.getSheetByName('Kanded');
    const mangijad = ss.getSheetByName('Mängijad');
    const harjutused = ss.getSheetByName('Harjutused');
    out.kanded = kanded ? sheetToObjects(kanded) : [];
    out.mangijad = mangijad ? sheetToObjects(mangijad) : [];
    out.harjutused = harjutused ? sheetToObjects(harjutused) : [];
    return ContentService.createTextOutput(JSON.stringify(out))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false, error:String(err)}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
